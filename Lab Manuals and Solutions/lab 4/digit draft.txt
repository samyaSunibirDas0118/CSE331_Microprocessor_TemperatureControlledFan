org 100h

.DATA

PROMPT DB "ENTER ANY CHARACTER: $"

CHAR DB 13,10,"ITS A CHARACTER!$"

DIGIT DB 13,10,"ITS A DIGIT!$"

.CODE

    MOV AX,@DATA
  
    MOV DS,AX
  
    ;PROMPT MESSAGE
  
    MOV DX,OFFSET PROMPT
  
    MOV AH,9
  
    INT 21H
  
    ;READ NUM/CHAR FROM THE KEYBOARD
  
    MOV AH,1
  
    INT 21H
  
    ;CHECK WHETHER THE INPUT IS LESS THAN 30H,THEN ITS A CHARACTER
  
    CMP AL,30H
  
    JL PRINTCHAR
  
    ;IF NOT LESS THAN 30H,CHECK WHETHER THE INPUT IS LESS THAN OR EQUAL TO 39H,THEN ITS A DIGIT
  
    CMP AL,39H
  
    JLE PRINTDIGIT
  
    ;ELSE ALL OTHER COMBINATIONS ARE CHARACTERS AND PRINT THE MESSAGE ITS A CHARACTER
  
PRINTCHAR: MOV DX,OFFSET CHAR

    MOV AH,9
  
    INT 21H
  
    JMP END
  
    ;PRINT THE MESSAGE ITS A DIGIT
  
PRINTDIGIT: MOV DX,OFFSET DIGIT

    MOV AH,9

    INT 21H
  
END: HLT

ret
